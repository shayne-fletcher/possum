1use crate::BoxError;
2use reqwest::Client;
3use serde_json::Value;

5/// Search for repositories by keyword(s) and optional filter; returns the
6/// matching `modelId`s.
7pub async fn search(
8    keywords: &[String],
9    filter: Option<&str>,
10    api_base_url: &str,
11) -> Result<Vec<String>, BoxError> {
12    let client = Client::new();
13    let url = build_search_url(keywords, filter, api_base_url);
14    let response = client.get(&url).send().await?;
15    if response.status().is_success() {
16        let models: Value = response.json().await?;
17        Ok(model_ids(&models))
18    } else {
19        let query = keywords.join(" ");
20        Err(format!(
21            "Failed to search models with keyword '{query}' (HTTP {})",
22            response.status()
23        )
24        .into())
25    }
26}
27
28/// Extract `modelId`s from a search response array.
29pub fn model_ids(models: &Value) -> Vec<String> {
30    models
31        .as_array()
32        .map(|list| {
33            list.iter()
34                .filter_map(|m| {
35                    m.get("modelId")
36                        .and_then(|id| id.as_str())
37                        .map(String::from)
38                })
39                .collect()
40        })
41        .unwrap_or_default()
42}
43
44pub fn build_search_url(keywords: &[String], filter: Option<&str>, api_base_url: &str) -> String {
45    let search_query = keywords.join(" ");
46    let mut url = format!("{api_base_url}/api/models?search={search_query}");
47    if let Some(f) = filter {
48        url.push_str(&format!("&filter={f}"));
49    }
50    url
51}
